def wiftoNATO():
    global NATO_characters, character, wif, NATO
    i = 0
    for x in NATO_characters:
        if x[0] == character:
            wif.append(x[0])
            NATO.append(x[1])
        else:
            i = i + 1
            if i >= len(NATO_characters):
                NATO.append("---ERROR---")
                wif.append("---ERROR---")
    return NATO
def check_NATO():
    global NATO_characters, word, wif, NATO
    i = 0
    check = False
    for x in NATO_characters:
        if x[1] == word:
            wif.append(x[0])
            NATO.append(x[1])
            check = True
        else:
            i = i + 1
            if i >= len(NATO_characters):
                check = False
    return check
